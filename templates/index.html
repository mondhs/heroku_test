            var word = $('#word').val();
            //alert(word);
            var resultListHtml = $('#result');
            word = word.replace(/[\s]+/g, "_");
            $.getJSON( "lieptas/api/v1.0/dictionary?words="+word, function( data ) {
                    var text =  "";//$('#result').text();
                    $.each( data, function( key, val ) {
                            //alert(val);
                            text = text + key + "\t" + val + "\n" ;
                    });
                     $('#result').text(text);
                    
            });
             $('#word').val("");
    });

        var save_this = $(this);

           save_this.select(); 
        },100);
    });
